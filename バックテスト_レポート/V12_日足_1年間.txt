"""
バックテスト v12: 1時間足・過去1年検証モデル
- タイムフレーム: 1時間足 (interval="1h")
- 期間: 過去1年間 (period="1y")
- 目的: 長期的な市場の波に対するロジックの堅牢性確認
"""

import pandas as pd
import yfinance as yf
import yaml
import pandas_ta as ta
from datetime import datetime, timedelta
from backtest.engine import BacktestEngine
from backtest.reporter import generate_report, plot_equity_curve
from backtest.screener import screen_stocks
from strategy.ensemble import EnsembleEngine

def load_hourly_data(ticker):
    """1時間足データを1年分取得"""
    # yfinanceの制限: 1hデータは過去730日まで取得可能
    data = yf.download(ticker, period="1y", interval="1h", progress=False)
    
    if data.empty: return None
    
    # カラム名整理
    if isinstance(data.columns, pd.MultiIndex):
        data.columns = [col[0].lower() for col in data.columns]
    else:
        data.columns = [col.lower() for col in data.columns]
        
    data = data[["open", "high", "low", "close", "volume"]].copy()
    data.dropna(inplace=True)
    
    # タイムゾーン変換 (UTC -> Asia/Tokyo)
    if data.index.tz is not None:
        data.index = data.index.tz_convert("Asia/Tokyo")
    else:
        data.index = data.index.tz_localize("UTC").tz_convert("Asia/Tokyo")
        
    return data

def load_daily_data(ticker):
    """日足データを2年分取得（1年間のバックテストに十分なバッファを持たせる）"""
    data = yf.download(ticker, period="2y", interval="1d", progress=False)
    if data.empty: return None
    if isinstance(data.columns, pd.MultiIndex):
        data.columns = [col[0].lower() for col in data.columns]
    else:
        data.columns = [col.lower() for col in data.columns]
    data = data[["open", "high", "low", "close", "volume"]].copy()
    data.dropna(inplace=True)
    return data

def calc_daily_bias(daily_df, config):
    """日足でのトレンド判定"""
    dc = config["daily_bias"]
    df = daily_df.copy()
    df["ema_s"] = ta.ema(df["close"], length=dc["ema_short"])
    df["ema_l"] = ta.ema(df["close"], length=dc["ema_long"])
    
    bias = {}
    for idx, row in df.iterrows():
        date = idx.date()
        if pd.isna(row["ema_s"]) or pd.isna(row["ema_l"]):
            bias[date] = "NEUTRAL"
        elif row["ema_s"] > row["ema_l"]:
            bias[date] = "BULL"
        else:
            bias[date] = "BEAR"
    return bias

def apply_timeframe_filter(signals_df, daily_bias):
    """日足トレンドと1時間足シグナルの整合性チェック"""
    result = signals_df.copy()
    filtered = 0
    for idx in result.index:
        date = idx.date()
        bias = daily_bias.get(date, "NEUTRAL")
        signal = result.loc[idx, "final_signal"]
        
        # BULLならSELL禁止、BEARならBUY禁止
        if bias == "BULL" and signal == "SELL":
            result.loc[idx, "final_signal"] = "HOLD"
            filtered += 1
        elif bias == "BEAR" and signal == "BUY":
            result.loc[idx, "final_signal"] = "HOLD"
            filtered += 1
            
    return result, filtered

def main():
    print("=" * 60)
    print("  📊 日本株バックテスト v12: 1時間足・過去1年検証")
    print("=" * 60)

    with open("config/strategy_config.yaml", "r", encoding="utf-8") as f:
        config = yaml.safe_load(f)

    # 1時間足でもスクリーニングを適用（直近のボラティリティが高い銘柄）
    print("\n■ 銘柄スクリーニング中...")
    selected = screen_stocks(config)
    tickers = [s["ticker"] for s in selected]

    if not tickers:
        print("銘柄が見つかりませんでした。")
        return

    print(f"\n■ データロード & 解析（対象: {len(tickers)}銘柄）...")
    ensemble = EnsembleEngine("config/strategy_config.yaml")
    signals_dict = {}
    
    total_processed = 0
    for ticker in tickers:
        try:
            df_1h = load_hourly_data(ticker)
            df_daily = load_daily_data(ticker)
            if df_1h is None or df_daily is None or len(df_1h) < 100:
                continue
            
            bias = calc_daily_bias(df_daily, config)
            signals_df = ensemble.generate_ensemble_signals(df_1h)
            signals_df, _ = apply_timeframe_filter(signals_df, bias)
            
            signals_dict[ticker] = signals_df
            total_processed += 1
            print(f"  [{total_processed}/{len(tickers)}] {ticker} 完了")
        except Exception as e:
            print(f"  ❌ {ticker} エラー: {e}")

    print(f"\n■ バックテストシミュレーション中（1年分）...")
    engine = BacktestEngine("config/strategy_config.yaml")
    result = engine.run(signals_dict)

    print("\n" + "=" * 60)
    report = generate_report(result, engine.initial_capital)
    print(report)

    if result.trades:
        # 取引日数を算出
        all_dates = set(t.entry_date.date() for t in result.trades)
        total_days = len(all_dates)
        if total_days > 0:
            print(f"  🔥 総取引回数: {len(result.trades)} 回")
            print(f"  🔥 1日平均取引数: {len(result.trades)/total_days:.2f} 回")
        
        longs = [t for t in result.trades if t.side.value == "LONG"]
        shorts = [t for t in result.trades if t.side.value == "SHORT"]
        print(f"  📈 ロング: {len(longs)}回 (勝率 {len([t for t in longs if t.pnl>0])/max(len(longs),1)*100:.1f}%)")
        print(f"  📉 ショート: {len(shorts)}回 (勝率 {len([t for t in shorts if t.pnl>0])/max(len(shorts),1)*100:.1f}%)")

    plot_equity_curve(result, engine.initial_capital, save_path="equity_curve_1year.png")
    print(f"\n✅ 1年分の検証完了。グラフ: equity_curve_1year.png")

if __name__ == "__main__":
    main()
